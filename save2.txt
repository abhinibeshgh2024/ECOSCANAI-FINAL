model = genai.GenerativeModel('gemini-3.1-flash-lite-preview')

def get_db():
    if not os.path.exists(DB_PATH): return {}
    with open(DB_PATH, "r") as f: return json.load(f)

def save_db(data):
    with open(DB_PATH, "w") as f: json.dump(data, f, indent=4)

@app.post("/api/analyze")
async def analyze_and_save(
    date: str = Form(...), 
    source: str = Form(...), 
    name: str = Form(None), 
    qty: str = Form(None), 
    file: UploadFile = File(None)
):
    prompt = """Return ONLY a JSON object: {"name": str, "carbon": float, "solution": str, "pkg": str, "saving": float, "efficiency": int} 
    Analyze the carbon footprint for: """
    
    try:
        if file:
            img_bytes = await file.read()
            response = model.generate_content([f"{prompt} this receipt image.", {"mime_type": "image/jpeg", "data": img_bytes}])
        else:
            response = model.generate_content(f"{prompt} {qty} servings of {name}.")

        # Cleaning response
        raw_json = re.search(r'\{.*\}', response.text, re.DOTALL).group()
        item = json.loads(raw_json)
        item.update({"id": str(uuid.uuid4())[:5], "source": source})

        db = get_db()
        if date not in db: db[date] = {"canteen": [], "nescafe": []}
        db[date][source].append(item)
        save_db(db)
        return item
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))

@app.get("/api/strategy")
async def get_ai_strategy():
    db = get_db()
    # Flatten all data for analysis
    all_items = []
    for date in db:
        all_items.extend(db[date]["canteen"])
        all_items.extend(db[date]["nescafe"])
    
    if not all_items:
        return {"strategy": "No data available yet. Start logging items to see AI insights."}

    summary = f"The university has logged {len(all_items)} items. Total carbon footprint recorded. Some items include {all_items[0]['name'] if all_items else ''}."
    
    prompt = f"""
    Act as a Sustainability Consultant. Based on this university food data: {summary}.
    1. Compare performance against UN SDG 12 (Responsible Consumption) and SDG 13 (Climate Action).
    2. Mention how top global institutes (like MIT or Oxford) reduce canteen waste.
    3. Provide 3 high-impact institutional policy changes.
    4. Estimate the percentage achieved towards 'Net Zero Campus' based on this data.
    Format the response in clear, concise bullet points.
    """
    
    try:
        response = model.generate_content(prompt)
        return {"strategy": response.text}
    except Exception as e:
        return {"strategy": "AI Strategy failed to generate."}

    <script>
        const API = "http://localhost:8000/api";
        let masterDB = {};
        let chart = null;

        document.getElementById('target-date').valueAsDate = new Date();
        syncUI();

        function showLoader(show) { document.getElementById('loader').style.display = show ? 'flex' : 'none'; }

        async function syncUI() {
            const res = await fetch(`${API}/db`);
            masterDB = await res.json();
            renderLiveFeed();
            updateCumulativeStats();
        }

        function nav(page) {
            document.querySelectorAll('[id^="page-"]').forEach(p => p.style.display = 'none');
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById('page-' + page).style.display = 'block';
            document.getElementById('btn-' + page).classList.add('active');
            if(page === 'stats') renderChart();
        }

        async function process(source) {
            showLoader(true);
            const date = document.getElementById('target-date').value;
            const fd = new FormData();
            fd.append('date', date); fd.append('source', source);

            if(source === 'canteen') {
                fd.append('name', document.getElementById('m-name').value);
                fd.append('qty', document.getElementById('m-qty').value);
            } else {
                const file = document.getElementById('n-file').files[0];
                if(!file) { showLoader(false); return alert("Select receipt"); }
                fd.append('file', file);
            }

            try {
                await fetch(`${API}/analyze`, { method: 'POST', body: fd });
                syncUI();
            } finally { showLoader(false); }
        }

        async function generateGlobalStrategy() {
            showLoader(true);
            try {
                const res = await fetch(`${API}/strategy`);
                const data = await res.json();
                document.getElementById('ai-strategy-content').innerHTML = `
                    <h2 style="color:var(--accent)">UN SDG Alignment Report</h2>
                    <p>${data.strategy.replace(/\n/g, '<br>')}</p>
                `;
            } catch (e) { alert("Strategy generation failed."); }
            finally { showLoader(false); }
        }

        function renderLiveFeed() {
            const date = document.getElementById('target-date').value;
            const feed = document.getElementById('live-feed');
            feed.innerHTML = "";
            if(masterDB[date]) {
                [...masterDB[date].canteen, ...masterDB[date].nescafe].reverse().forEach(item => {
                    feed.innerHTML += `
                        <div class="item-card ${item.source === 'nescafe' ? 'nescafe-border' : ''}">
                            <div style="display:flex; justify-content:space-between;">
                                <strong>${item.name}</strong>
                                <span style="color:var(--accent)">${item.carbon} kg CO2e</span>
                            </div>
                            <div style="font-size:0.85rem; margin-top:8px; color:#94a3b8;">
                                Swap: ${item.solution} | Score: ${item.efficiency}%
                            </div>
                        </div>`;
                });
            }
        }

        function updateCumulativeStats() {
            let total = 0, saved = 0;
            Object.values(masterDB).forEach(day => {
                [...day.canteen, ...day.nescafe].forEach(i => { total += i.carbon; saved += i.saving; });
            });
            document.getElementById('s-daily').innerText = total.toFixed(1) + "kg";
            document.getElementById('s-eff').innerText = ((saved / (total + 1)) * 100).toFixed(0) + "%";
            document.getElementById('s-gap').innerText = total > 50 ? "Over Target" : "On Track";
        }

        function renderChart() {
            let cant = 0, nesc = 0;
            Object.values(masterDB).forEach(day => {
                day.canteen.forEach(i => cant += i.carbon);
                day.nescafe.forEach(i => nesc += i.carbon);
            });
            const ctx = document.getElementById('compChart').getContext('2d');
            if(chart) chart.destroy();
            chart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Canteen', 'Nescafe'],
                    datasets: [{ data: [cant, nesc], backgroundColor: ['#10b981', '#f59e0b'], borderWeight: 0 }]
                },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { position: 'bottom', labels: { color: 'white' } } } }
            });
        }

        async function resetSystem() { if(confirm("Clear all data?")) { await fetch(`${API}/clear`, { method: 'DELETE' }); syncUI(); } }
