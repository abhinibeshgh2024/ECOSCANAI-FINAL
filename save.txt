        const API = "http://localhost:8000/api";

        syncUI();

        async function syncUI() {
            const res = await fetch(`${API}/db`);
            masterDB = await res.json();

            renderHistory();

        function nav(page) {
            document.querySelectorAll('[id^="page-"]').forEach(p => p.style.display = 'none');
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById('page-' + page).style.display = 'block';
            document.getElementById('btn-' + page).classList.add('active');
            if(page === 'stats') renderChart();

        async function process(source) {
            const date = document.getElementById('target-date').value;
            const fd = new FormData();
            fd.append('date', date); fd.append('source', source);

            if(source === 'canteen') {
                fd.append('name', document.getElementById('m-name').value);
                fd.append('qty', document.getElementById('m-qty').value);
            } else {
                const file = document.getElementById('n-file').files[0];
                if(!file) return alert("Select receipt");
                fd.append('file', file);
            }

            const res = await fetch(`${API}/analyze`, { method: 'POST', body: fd });
            if(res.ok) {
                alert("AI Analysis Complete!");
                syncUI();
            }
        }

        function renderLiveFeed() {
            const date = document.getElementById('target-date').value;
            const feed = document.getElementById('live-feed');
            feed.innerHTML = "";
            
            if(masterDB[date]) {
                [...masterDB[date].canteen, ...masterDB[date].nescafe].reverse().forEach(item => {
                    feed.innerHTML += `
                        <div class="item-card ${item.source === 'nescafe' ? 'nescafe-border' : ''}">
                            <div style="display:flex; justify-content:space-between;">
                                <strong>${item.name} (${item.source.toUpperCase()})</strong>
                                <span style="color:var(--accent)">${item.carbon} kg CO2e</span>
                            </div>
                            <div style="font-size:0.85rem; margin-top:8px; color:#94a3b8;">
                                🌿 <b>Swap:</b> ${item.solution} | 📦 <b>Pkg:</b> ${item.pkg} | 📈 <b>Score:</b> ${item.efficiency}%
                            </div>
                        </div>
                    `;
                });
            }
        }

        function renderHistory() {
            const container = document.getElementById('history-container');
            container.innerHTML = "";
            Object.keys(masterDB).sort().reverse().forEach(date => {
                const day = masterDB[date];
                const total = [...day.canteen, ...day.nescafe].reduce((sum, i) => sum + i.carbon, 0);
                container.innerHTML += `
                    <div class="glass-panel">
                        <div style="display:flex; justify-content:space-between; border-bottom:1px solid #334155; padding-bottom:10px; margin-bottom:10px;">
                            <h3 style="margin:0">${date}</h3>

        function jumpToDate(date) {
            document.getElementById('target-date').value = date;
            nav('entry');
            syncUI();
        }

        function updateCumulativeStats() {
            const today = document.getElementById('target-date').value;
            let dTotal = 0, wTotal = 0, yTotal = 0, sTotal = 0, cTotal = 0;
            const now = new Date();

            Object.keys(masterDB).forEach(date => {
                const d = new Date(date);
                const items = [...masterDB[date].canteen, ...masterDB[date].nescafe];
                const dayCarbon = items.reduce((sum, i) => sum + i.carbon, 0);
                const daySaved = items.reduce((sum, i) => sum + i.saving, 0);

                if(date === today) dTotal += dayCarbon;
                if(now - d < 7 * 24 * 3600000) wTotal += dayCarbon;
                if(d.getFullYear() === now.getFullYear()) yTotal += dayCarbon;
                sTotal += daySaved;
                cTotal += dayCarbon;
            });

            document.getElementById('s-daily').innerText = dTotal.toFixed(1);
            document.getElementById('s-weekly').innerText = wTotal.toFixed(1);
            document.getElementById('s-yearly').innerText = yTotal.toFixed(1);
            document.getElementById('s-eff').innerText = ((sTotal / (cTotal + 1)) * 100).toFixed(0) + "%";
        }

        function renderChart() {
            let cant = 0, nesc = 0;
            Object.values(masterDB).forEach(day => {
                day.canteen.forEach(i => cant += i.carbon);
                day.nescafe.forEach(i => nesc += i.carbon);
            });

            const ctx = document.getElementById('compChart').getContext('2d');
            if(chart) chart.destroy();
            chart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Main Canteen', 'Nescafe Outlet'],
                    datasets: [{
                        label: 'Total Emission (kg CO2e)',
                        data: [cant, nesc],
                        backgroundColor: ['#10b981', '#f59e0b'],
                        borderRadius: 8
                    }]
                },
                options: { 
                    responsive: true, maintainAspectRatio: false,
                    plugins: { legend: { labels: { color: 'white' } } },
                    scales: { y: { ticks: { color: '#94a3b8' }, grid: { color: 'rgba(255,255,255,0.05)' } } }
                }
            });
        }

        async function resetSystem() {
            if(confirm("Delete ALL historical data?")) {
                await fetch(`${API}/clear`, { method: 'DELETE' });
                syncUI();
            }
        }

def get_db():
    if not os.path.exists(DB_PATH): return {}

def save_db(data):
    with open(DB_PATH, "w") as f: json.dump(data, f, indent=4)

@app.post("/api/analyze")
async def analyze_and_save(
    date: str = Form(...), 
    source: str = Form(...), 
    name: str = Form(None), 
    qty: str = Form(None), 
    file: UploadFile = File(None)
):
    # Strict System Instruction for AI
    prompt = f"""Analyze the environmental impact of {source} items. 
    Return ONLY a JSON list of objects with these keys: 
    "name", "carbon" (kg CO2, float), "solution" (specific sustainable swap), 
    "pkg" (packaging type), "saving" (potential kg CO2 saved with swap, float), 
    "efficiency" (sustainability score 1-100, int)."""

    try:
        if file:
            img_bytes = await file.read()
            response = model.generate_content([f"{prompt} Source: {source} receipt.", {"mime_type": "image/jpeg", "data": img_bytes}])
        else:
            response = model.generate_content(f"{prompt} Item: {qty} of {name}.")

        # Robust JSON extraction
        clean_text = re.sub(r'```json|```', '', response.text).strip()
        items = json.loads(clean_text)
        if isinstance(items, dict): items = [items]

        db = get_db()
        if date not in db: db[date] = {"canteen": [], "nescafe": []}
        
        processed_items = []
        for item in items:
            item.update({
                "id": str(uuid.uuid4())[:8],
                "timestamp": datetime.now().strftime("%H:%M:%S"),
                "source": source
            })
            db[date][source].append(item)
            processed_items.append(item)
        
        save_db(db)
        return processed_items
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"AI Analysis Failed: {str(e)}")

@app.get("/api/db")
