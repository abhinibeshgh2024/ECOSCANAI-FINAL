    <script>
        const { jsPDF } = window.jspdf;
        const API = "http://localhost:8000/api";
        let masterDB = {};
        let chart = null;

        document.getElementById('target-date').valueAsDate = new Date();
        syncUI();

        function showLoader(s) { document.getElementById('loader').style.display = s ? 'flex' : 'none'; }

        async function syncUI() {
            const res = await fetch(`${API}/db`);
            masterDB = await res.json();
            renderFeed();
            updateDashboard();
        }

        function nav(p) {
            document.querySelectorAll('[id^="page-"]').forEach(el => el.style.display = 'none');
            document.querySelectorAll('.nav-btn').forEach(el => el.classList.remove('active'));
            document.getElementById('page-' + p).style.display = 'block';
            document.getElementById('btn-' + p).classList.add('active');
            if(p === 'stats') setTimeout(renderChart, 100);
        }

        async function process(source) {
            showLoader(true);
            const date = document.getElementById('target-date').value;
            const fd = new FormData();
            fd.append('date', date); fd.append('source', source);
            if(source === 'canteen') {
                fd.append('name', document.getElementById('m-name').value);
                fd.append('qty', document.getElementById('m-qty').value);
            } else {
                const file = document.getElementById('n-file').files[0];
                if(!file) { showLoader(false); return alert("Upload file first"); }
                fd.append('file', file);
            }
            try { await fetch(`${API}/analyze`, { method: 'POST', body: fd }); syncUI(); } 
            finally { showLoader(false); }
        }

        async function runSimulation() {
            const scenario = document.getElementById('sim-input').value;
            if(!scenario) return;
            showLoader(true);
            try {
                const res = await fetch(`${API}/simulate`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ scenario })
                });
                const data = await res.json();
                const resDiv = document.getElementById('sim-result');
                resDiv.style.display = 'block';
                resDiv.innerText = data.prediction; 
            } finally { showLoader(false); }
        }

        function updateDashboard() {
            let total = 0, saved = 0;
            Object.values(masterDB).forEach(day => {
                [...day.canteen, ...day.nescafe].forEach(i => { total += i.carbon; saved += i.saving; });
            });
            document.getElementById('s-daily').innerText = total.toFixed(1) + " kg";
            document.getElementById('s-energy').innerText = (total * 2.5).toFixed(1) + " kWh";
            document.getElementById('s-bulbs').innerText = Math.floor((total * 2.5)/0.01).toLocaleString() + " hrs";
            document.getElementById('s-eff').innerText = ((saved / (total + 1)) * 100).toFixed(0) + "%";
        }

        function renderFeed() {
            const date = document.getElementById('target-date').value;
            const feed = document.getElementById('live-feed');
            feed.innerHTML = "";
            if(masterDB[date]) {
                [...masterDB[date].canteen, ...masterDB[date].nescafe].reverse().forEach(i => {
                    feed.innerHTML += `
                        <div class="item-row" style="border-color:${i.source==='canteen'?'#10b981':'#f59e0b'}">
                            <div><strong>${i.name}</strong><br><small>${i.solution}</small></div>
                            <div style="text-align:right"><strong>${i.carbon}kg</strong></div>
                        </div>`;
                });
            }
        }

        function renderChart() {
            let c = 0, n = 0;
            Object.values(masterDB).forEach(day => {
                day.canteen.forEach(i => c += i.carbon);
                day.nescafe.forEach(i => n += i.carbon);
            });
            const ctx = document.getElementById('compChart').getContext('2d');
            if(chart) chart.destroy();
            chart = new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels: ['Main Canteen', 'Nescafe'],
                    datasets: [{ data: [c, n], backgroundColor: ['#10b981', '#f59e0b'], borderWidth: 0 }]
                },
                options: { responsive: true, maintainAspectRatio: false, plugins: { legend: { position: 'bottom', labels: { color: 'white' } } } }
            });
        }

        function generateDailyPDF() {
            const date = document.getElementById('target-date').value;
            if(!masterDB[date]) return;
            const doc = new jsPDF();
            doc.text(`Eco-Scan Daily Report: ${date}`, 14, 20);
            const rows = [...masterDB[date].canteen, ...masterDB[date].nescafe].map(i => [i.name, i.source.toUpperCase(), i.carbon + 'kg', i.pkg]);
            doc.autoTable({ head: [['Item', 'Outlet', 'Carbon', 'Packaging']], body: rows, startY: 30 });
            doc.save(`Daily_Log_${date}.pdf`);
        }

        function generateAuditPDF() {
            const doc = new jsPDF();
            doc.setFontSize(18); doc.text("Campus Sustainability Audit Report", 14, 20);
            doc.setFontSize(10); doc.text("Generated by Eco-Scan AI Analytical Engine", 14, 28);
            
            let startY = 40;
            Object.keys(masterDB).forEach(date => {
                doc.setFontSize(12); doc.text(`Date: ${date}`, 14, startY);
                const rows = [...masterDB[date].canteen, ...masterDB[date].nescafe].map(i => [i.name, i.carbon + 'kg', i.solution]);
                doc.autoTable({ body: rows, startY: startY + 5, theme: 'grid', styles: { fontSize: 8 } });
                startY = doc.lastAutoTable.finalY + 15;
                if(startY > 260) { doc.addPage(); startY = 20; }
            });
            doc.save("University_Full_Audit.pdf");
        }

        async function resetSystem() { if(confirm("Erase all data?")) { await fetch(`${API}/clear`, { method: 'DELETE' }); location.reload(); } }
    </script>
