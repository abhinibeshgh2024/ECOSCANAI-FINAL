model = genai.GenerativeModel('gemini-3.1-flash-lite-preview')

class SimulationRequest(BaseModel):
    scenario: str

def get_db():
    if not os.path.exists(DB_PATH): return {}
    with open(DB_PATH, "r") as f: 
        try: return json.load(f)
        except: return {}

def save_db(data):
    with open(DB_PATH, "w") as f: json.dump(data, f, indent=4)

@app.post("/api/analyze")
async def analyze_and_save(
    date: str = Form(...), 
    source: str = Form(...), 
    name: str = Form(None), 
    qty: str = Form(None), 
    file: UploadFile = File(None)
):
    # Strict instruction to avoid symbols in suggestions
    prompt = """Analyze environmental impact. Return ONLY a JSON object: 
    {"name": str, "carbon": float, "solution": str, "pkg": str, "saving": float, "efficiency": int}
    Ensure the solution string is a formal sentence without any symbols or asterisks."""

    try:
        if file:
            img_bytes = await file.read()
            response = model.generate_content([f"{prompt} Source: {source} receipt.", {"mime_type": "image/jpeg", "data": img_bytes}])
        else:
            response = model.generate_content(f"{prompt} Item: {qty} of {name} from {source}.")

        clean_text = re.search(r'\{.*\}', response.text, re.DOTALL).group()
        item = json.loads(clean_text)
        
        item.update({
            "id": str(uuid.uuid4())[:8],
            "timestamp": datetime.now().strftime("%H:%M:%S"),
            "source": source
        })

        db = get_db()
        if date not in db: db[date] = {"canteen": [], "nescafe": []}
        db[date][source].append(item)
        save_db(db)
        return item
    except Exception as e:
        raise HTTPException(status_code=500, detail=f"AI Error: {str(e)}")

@app.post("/api/simulate")
async def simulate_scenario(request: SimulationRequest):
    db = get_db()
    # Prompt explicitly forbids symbols and markdown formatting
    prompt = f"""
    Analyze this policy change: "{request.scenario}"
    Provide a formal analysis in plain paragraphs. 
    Do not use asterisks, bullet point symbols, or hashtags. 
    Use clear headings like Carbon Reduction Analysis, United Nations Alignment, and Difficulty Assessment.
    Keep the language academic and professional.
    """
    try:
        response = model.generate_content(prompt)
        return {"prediction": response.text}
    except Exception as e:
        raise HTTPException(status_code=500, detail="Simulation Failed")

@app.get("/api/db")
async def fetch_db(): return get_db()

@app.delete("/api/clear")
async def clear(): 
    save_db({})
    return {"status": "Wiped"}

    <script>
        const { jsPDF } = window.jspdf;
        const API = "http://localhost:8000/api";
        let masterDB = {};
        let chart = null;

        document.getElementById('target-date').valueAsDate = new Date();
        syncUI();

        function showLoader(state) { document.getElementById('loader').style.display = state ? 'flex' : 'none'; }

        async function syncUI() {
            const res = await fetch(`${API}/db`);
            masterDB = await res.json();
            renderLiveFeed();
            updateStats();
        }

        function nav(page) {
            document.querySelectorAll('[id^="page-"]').forEach(p => p.style.display = 'none');
            document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
            document.getElementById('page-' + page).style.display = 'block';
            document.getElementById('btn-' + page).classList.add('active');
            if(page === 'stats') setTimeout(renderChart, 100);
        }

        async function process(source) {
            showLoader(true);
            const date = document.getElementById('target-date').value;
            const fd = new FormData();
            fd.append('date', date); fd.append('source', source);
            if(source === 'canteen') {
                fd.append('name', document.getElementById('m-name').value);
                fd.append('qty', document.getElementById('m-qty').value);
            } else {
                const file = document.getElementById('n-file').files[0];
                if(!file) { showLoader(false); return alert("Attach file"); }
                fd.append('file', file);
            }
            try { await fetch(`${API}/analyze`, { method: 'POST', body: fd }); syncUI(); } 
            finally { showLoader(false); }
        }

        async function runSimulation() {
            const scenario = document.getElementById('sim-input').value;
            if(!scenario) return;
            showLoader(true);
            try {
                const res = await fetch(`${API}/simulate`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ scenario })
                });
                const data = await res.json();
                const box = document.getElementById('sim-result');
                box.style.display = 'block';
                box.innerText = data.prediction; // Using innerText to ensure no HTML/Markdown rendering
            } finally { showLoader(false); }
        }

        function updateStats() {
            let total = 0, saved = 0;
            Object.values(masterDB).forEach(day => {
                [...day.canteen, ...day.nescafe].forEach(i => { total += i.carbon; saved += i.saving; });
            });
            document.getElementById('s-daily').innerText = total.toFixed(1) + " kg";
            document.getElementById('s-energy').innerText = (total * 2.5).toFixed(1) + " kWh";
            document.getElementById('s-bulbs').innerText = Math.floor((total * 2.5)/0.01).toLocaleString() + " hrs";
            document.getElementById('s-eff').innerText = ((saved / (total + 1)) * 100).toFixed(0) + "%";
        }

        function renderLiveFeed() {
            const date = document.getElementById('target-date').value;
            const feed = document.getElementById('live-feed');
            feed.innerHTML = "";
            if(masterDB[date]) {
                [...masterDB[date].canteen, ...masterDB[date].nescafe].reverse().forEach(i => {
                    feed.innerHTML += `
                        <div class="item-card" style="border-left-color: ${i.source==='canteen'?'#10b981':'#f59e0b'}">
                            <div style="display:flex; justify-content:space-between">
                                <span style="font-weight:bold">${i.name}</span>
                                <span style="color:var(--accent)">${i.carbon}kg CO2e</span>
                            </div>
                            <p style="margin:10px 0 0; font-size:0.9rem; color:#94a3b8">${i.solution}</p>
                        </div>`;
                });
            }
        }

        function renderChart() {
            let c = 0, n = 0;
            Object.values(masterDB).forEach(day => {
                day.canteen.forEach(i => c += i.carbon);
                day.nescafe.forEach(i => n += i.carbon);
            });
            const ctx = document.getElementById('compChart').getContext('2d');
            if(chart) chart.destroy();
            chart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Main Canteen', 'Nescafe'],
                    datasets: [{ label: 'kg CO2e', data: [c, n], backgroundColor: ['#10b981', '#f59e0b'] }]
                },
                options: { responsive: true, maintainAspectRatio: false, scales: { y: { grid: { color: '#334155' } } } }
            });
        }

        // PDF GENERATION LOGIC
        function downloadDailyPDF() {
            const date = document.getElementById('target-date').value;
            if(!masterDB[date]) return alert("No data for this date");
            const doc = new jsPDF();
            doc.setFontSize(20); doc.text(`Eco-Scan Daily Log: ${date}`, 14, 22);
            const items = [...masterDB[date].canteen, ...masterDB[date].nescafe].map(i => [i.name, i.source, i.carbon, i.pkg]);
            doc.autoTable({ head: [['Item', 'Source', 'CO2 (kg)', 'Packaging']], body: items, startY: 30 });
            doc.save(`EcoScan_Daily_${date}.pdf`);
        }

        function downloadFullAudit() {
            const doc = new jsPDF();
            doc.setFontSize(22); doc.text("Institutional Sustainability Audit", 14, 25);
            doc.setFontSize(12); doc.text("Generated by Eco-Scan AI Core Intelligence", 14, 35);
            
            let y = 50;
            Object.keys(masterDB).forEach(date => {
                const total = [...masterDB[date].canteen, ...masterDB[date].nescafe].reduce((s, i) => s + i.carbon, 0);
                doc.setFontSize(14); doc.text(`Date: ${date} | Total Emissions: ${total.toFixed(2)}kg`, 14, y);
                const items = [...masterDB[date].canteen, ...masterDB[date].nescafe].map(i => [i.name, i.carbon + 'kg', i.solution]);
                doc.autoTable({ body: items, startY: y + 5, styles: { fontSize: 8 } });
                y = doc.lastAutoTable.finalY + 15;
                if (y > 250) { doc.addPage(); y = 20; }
            });
            doc.save("University_Sustainability_Audit.pdf");
        }
